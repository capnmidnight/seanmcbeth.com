Make a custom cursor
//A frame on which we want to change the cursor
JFrame window = new JFrame("Different Cursor");
window.setSize(400, 300);
window.setVisible(true);

/*
 *... a transparent cursor
 */
BufferedImage cursor = new BufferedImage(16, 16, BufferedImage.TYPE_INT_ARGB);

/*
 *... or from a file
 */
BufferedImage cursor = javax.imageio.ImageIO.read(new File(""));

//create the cursor
Cursor transparentCursor = Toolkit.getDefaultToolkit().createCustomCursor(cursor, new Point(0, 0),"TransparentCursor");
//set the cursor
window.setCursor(transparentCursor);
//enjoy!
